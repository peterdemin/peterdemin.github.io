iptables -t mangle -I OUTPUT -m owner --uid-owner deluge -j MARK --set-mark 42

iptables -t nat -I POSTROUTING -o tun0 -j MASQUERADE
ip rule add fwmark 42 table 42

for f in /proc/sys/net/ipv4/conf/*/rp_filter; do

    done;
